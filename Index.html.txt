<script>

let xp = Number(localStorage.getItem("xp")||0);
let day = Number(localStorage.getItem("day")||1);
let streak = Number(localStorage.getItem("streak")||0);
let lastDate = localStorage.getItem("lastDate") || "";
let completedToday = false;

function save(){
localStorage.setItem("xp",xp);
localStorage.setItem("day",day);
localStorage.setItem("streak",streak);
localStorage.setItem("lastDate",lastDate);
}

/* DAILY RESET */
function checkDay(){
let today = new Date().toDateString();

if(lastDate !== today){
day++;
streak++;
lastDate = today;
completedToday = false;
}
}

/* TASK SYSTEM */
function task(value, box){

if(box.checked){
xp += value;
}else{
xp -= value;
}

update();
}

/* UPDATE SYSTEM */
function update(){

checkDay();

let level="Rookie";
if(xp>10000) level="Built Different";
else if(xp>5000) level="Beast";
else if(xp>2000) level="Grinder";

let progress = Math.min((day/75)*100,100);

document.getElementById("xp").innerText=xp;
document.getElementById("day").innerText=day;
document.getElementById("streak").innerText=streak;
document.getElementById("progress").innerText=Math.floor(progress);
document.getElementById("bar").style.width=progress+"%";

let coach="Lock in.";
if(xp<0) coach="You're slipping.";
else if(streak>5) coach="You're on fire 🔥";
else if(progress>50) coach="Halfway. Don't slow down.";

document.getElementById("coach").innerText=coach;

save();
}

/* MUSIC */
function openMusic(){
window.open("https://music.apple.com","_blank");
}

/* PHOTO SAVE */
function savePhoto(e){
let file=e.target.files[0];
localStorage.setItem("photoName",file.name);
document.getElementById("photoMsg").innerText="Photo saved ✔";
}

update();

</script>
